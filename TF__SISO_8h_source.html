99  inline static TooN::Vector<> simplifyNumerator(const TooN::Vector<>& num_coeff, const TooN::Vector<>& den_coeff)
100  {
101  if (den_coeff[0] == 0)
102  {
103  throw std::domain_error("[TF_SISO::simplifyNumerator] Division by 0");
104  }
105 
106  if (num_coeff.size() == 0) // case void vector
107  {
108  return TooN::makeVector(0.0);
109  }
110 
111  // find first nonzero coeff
112  unsigned int effective_size = num_coeff.size();
113  while (effective_size != 1)
114  {
115  if (num_coeff[effective_size - 1] != 0.0)
116  break;
117  effective_size--;
118  }
119 
120  return num_coeff.slice(0, effective_size) / den_coeff[0];
121  }

130  inline static TooN::Vector<> simplifyAndReduceDenominator(const TooN::Vector<>& den_coeff)
131  {
132  if (den_coeff[0] == 0)
133  {
134  throw std::domain_error("[TF_SISO::simplifyAndReduceDenominator] Division by 0");
135  }
136 
137  // find first nonzero coeff
138  unsigned int effective_size = den_coeff.size();
139  while (effective_size != 1)
140  {
141  if (den_coeff[effective_size - 1] != 0.0)
142  break;
143  effective_size--;
144  }
145 
146  if (effective_size == 1)
147  {
148  return TooN::Vector<>(0); // return a void vector
149  }
150 
151  return den_coeff.slice(1, effective_size - 1) / den_coeff[0];
152  }

262  inline virtual double apply(double u_k) override
263  {
264  // Shift u_vec
265  const unsigned int no = u_vec_.size() - 1;
266  if (no != 0)
267  {
268  TooN::Vector<> tmp_u_vec = u_vec_;
269  u_vec_.slice(1, no) = tmp_u_vec.slice(0, no);
270  }
271  u_vec_[0] = u_k;
272 
273  // Shift y_vec
274  const unsigned int deno = y_vec_.size();
275  if (deno == 0)
276  {
277  y_k_[0] = b_vec_ * u_vec_;
278  return y_k_[0];
279  }
280  if (deno != 1)
281  {
282  TooN::Vector<> tmp_y_vec = y_vec_;
283  y_vec_.slice(1, deno - 1) = tmp_y_vec.slice(0, deno - 1);
284  }
285  y_vec_[0] = y_k_[0];
286 
287  y_k_[0] = (b_vec_ * u_vec_) - (a_vec_ * y_vec_);
288 
289  return y_k_[0];
290  }
