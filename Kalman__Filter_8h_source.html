179  inline virtual const TooN::Vector<>& obs_apply(const TooN::Vector<>& u_k, const TooN::Vector<>& y_k) override
180  {
181 
182 // TODO: review symbolism https://en.wikipedia.org/wiki/Extended_Kalman_filter
183 
184 #define u_k1 u_k
185 
186 #define x_hat_k_k state_
187 #define x_hat_k1_k1 state_
188 
189 #define P_k_k P_
190 #define P_k1_k1 P_
191 
192 #define W_k1 W_
193 #define V_k V_
194 
195 #define y_hat_k_k output_
196 
197  // x_hat_k1_k1 = x_hat_k_k;
198  // P_k1_k1 = P_k_k;
199 
200  /*PREDICT*/
201  // predict state estimate
202  TooN::Vector<> x_hat_k_k1 = system_->state_fcn(x_hat_k1_k1, u_k1);
203 
204  // Predicted covariance estimate
205  TooN::Matrix<> F_k1 = system_->jacob_state_fcn(x_hat_k1_k1, u_k1);
206  TooN::Matrix<> P_k_k1 = F_k1 * P_k1_k1 * (F_k1.T()) + W_k1;
207 
208  /*UPDATE*/
209  // Innovation or measurement residual
210  TooN::Vector<> y_hat_k_k1 = system_->output_fcn(x_hat_k_k1, u_k1);
211  TooN::Vector<> y_tilde_k = y_k - y_hat_k_k1;
212 
213  // Innovation (or residual) covariance
214  TooN::Matrix<> H_k = system_->jacob_output_fcn(x_hat_k_k1, u_k1);
215  TooN::Matrix<> S_k = H_k * P_k_k1 * (H_k.T()) + V_k;
216 
217  // Near-optimal Kalman gain
218  TooN::SVD<> S_k_SVD(S_k);
219  TooN::Matrix<> K_k = P_k_k1 * (H_k.T()) * S_k_SVD.get_pinv();
220 
221  // Update state estimate
222  x_hat_k_k = x_hat_k_k1 + K_k * y_tilde_k;
223 
224  // Update covariance estimate
225  P_k_k = (Identity_x_ - K_k * H_k) * P_k_k1;
226 
227  // Update Output
228  y_hat_k_k = system_->output_fcn(x_hat_k_k, u_k1);
229 
230  return y_hat_k_k;
231 
232 #undef x_hat_k_k
233 #undef x_hat_k1_k1
234 #undef P_k_k
235 #undef P_k1_k1
236 #undef W_k1
237 #undef V_k
238  }
