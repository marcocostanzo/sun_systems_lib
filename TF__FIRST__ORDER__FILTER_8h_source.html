87 static TooN::Vector<2> tf_first_order_get_num_coeff(double cut_freq, double Ts)
88 {
89  double tau = 1.0/(2.0*M_PI*cut_freq);
90  double _alpha = 2.0*tau/Ts;
91  return (1.0/(1.0+_alpha)) * TooN::makeVector( 1.0 , 1.0 );
92 }
93 
95 static TooN::Vector<2> tf_first_order_get_den_coeff(double cut_freq, double Ts)
96 {
97  double tau = 1.0/(2.0*M_PI*cut_freq);
98  double _alpha = 2.0*tau/Ts;
99  return TooN::makeVector( 1.0, ( (1.0-_alpha)/(1.0+_alpha) ) );
100 }
