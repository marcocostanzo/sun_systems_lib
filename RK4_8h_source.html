135  inline virtual const TooN::Vector<> state_fcn(const TooN::Vector<>& x_n_1, const TooN::Vector<>& u_n,
136  const TooN::Vector<>& u_n_1) const
137  {
138  TooN::Vector<> u_n_12 = estimateMeanInputs(u_n, u_n_1);
139 
140  TooN::Vector<> k1 = system_->state_fcn(x_n_1, u_n_1);
141  TooN::Vector<> k2 = system_->state_fcn(x_n_1 + Ts_2_ * k1, u_n_12);
142  TooN::Vector<> k3 = system_->state_fcn(x_n_1 + Ts_2_ * k2, u_n_12);
143  TooN::Vector<> k4 = system_->state_fcn(x_n_1 + Ts_ * k3, u_n);

171  inline virtual const TooN::Matrix<> jacob_state_fcn(const TooN::Vector<>& x_n_1, const TooN::Vector<>& u_n,
172  const TooN::Vector<>& u_n_1) const
173  {
174  TooN::Vector<> u_n_12 = estimateMeanInputs(u_n, u_n_1);
175 
176  TooN::Vector<> k1 = system_->state_fcn(x_n_1, u_n_1);
177  TooN::Vector<> k2 = system_->state_fcn(x_n_1 + Ts_2_ * k1, u_n_12);
178  TooN::Vector<> k3 = system_->state_fcn(x_n_1 + Ts_2_ * k2, u_n_12);
179 
180  TooN::Matrix<> jac_k1 = system_->jacob_state_fcn(x_n_1, u_n_1);
181  TooN::Matrix<> jac_k2 = system_->jacob_state_fcn(x_n_1 + Ts_2_ * k1, u_n_12) * (Identity_x_ + Ts_2_ * jac_k1);
182  TooN::Matrix<> jac_k3 = system_->jacob_state_fcn(x_n_1 + Ts_2_ * k2, u_n_12) * (Identity_x_ + Ts_2_ * jac_k2);
183  TooN::Matrix<> jac_k4 = system_->jacob_state_fcn(x_n_1 + Ts_ * k3, u_n) * (Identity_x_ + Ts_ * jac_k3);
184 
185  return Identity_x_ + Ts_6_ * (jac_k1 + 2.0 * jac_k2 + 2.0 * jac_k3 + jac_k4); //=jac_n
186  }
